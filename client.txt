# Lawvriksh Backend API - Example Requests (http://127.0.0.1:8000/)

# 1. Admin Login
http -f POST http://127.0.0.1:8000/adminlogin username==sahilsaurav2507@gmail.com password==Lawvriksh@2507//

# 2. Register User
http POST http://127.0.0.1:8000/userdata name='Test User' email='test@user.com' phone='1234567890'

# 3. Register Creator
http POST http://127.0.0.1:8000/creatordata name='Test Creator' email='creator@ex.com' phone='9876543210'

# 4. Not Interested
http POST http://127.0.0.1:8000/notinteresteddata email='not@ex.com' reason='Not interested'

# 5. Feedback
http POST http://127.0.0.1:8000/feedback contact_email='fb@ex.com' feedback:='{"message": "Great!"}'

# 6. Get Registered Users (Admin)
# (Replace <token> with the access_token from /adminlogin)
http GET http://127.0.0.1:8000/registereduserdata "Authorization: Bearer <token>"

# 7. Get Registered Creators (Admin)
http GET http://127.0.0.1:8000/registeredcreatordata "Authorization: Bearer <token>"

# 8. Download Data (Admin)
http -f POST http://127.0.0.1:8000/downloaddata table=users "Authorization: Bearer <token>"
